ðŸ§‘â€ðŸ’» Java Solution â€” Recursive
class Solution {
    public int subtractProductAndSum(int n) {
        return prod(n) - sum(n);
    }

    // Recursive product of digits
    public int prod(int n) {
        if (n == 0) return 1;
        return prod(n / 10) * (n % 10);
    }

    // Recursive sum of digits
    public int sum(int n) {
        if (n == 0) return 0;
        return sum(n / 10) + (n % 10);
    }
}

class Solution {
    public int subtractProductAndSum(int n) {
        int product = 1;
        int sum = 0;

        while (n > 0) {
            int digit = n % 10;
            product *= digit;
            sum += digit;
            n /= 10;
        }

        return product - sum;
    }
}
